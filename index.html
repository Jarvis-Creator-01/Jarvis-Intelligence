async function processQuery(text) { status.innerText = "Searching Global Brain..."; let response = ""; // 1. Personal Memory & Goal Check if (text.includes("billionaire") || text.includes("sapna")) { response = "Sir, aapka lakshya 19-20 saal ki umar mein youngest billionaire banna hai. Main aapke saath hoon."; } else { // Cleaning the search topic let topic = text.replace("jarvis", "") .replace("kaun hai", "") .replace("kaun tha", "") .replace("kya hai", "") .replace("batao", "") .replace("ke baare mein", "") .trim(); if(topic.length > 0) { // Priority 1: Hindi Wikipedia const hindiUrl = `https://hi.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(topic)}`; // Priority 2: English Wikipedia (Backup) const englishUrl = `https://en.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(topic)}`; try { let res = await fetch(hindiUrl); let data = await res.json(); if (data.extract) { response = data.extract; } else { // Agar Hindi mein nahi mila, English check karo let resEn = await fetch(englishUrl); let dataEn = await resEn.json(); if (dataEn.extract) { response = "Sir, Hindi records mein nahi mila par English records ke mutabik: " + dataEn.extract; } else { response = "Sir, '" + topic + "' ke baare mein 1000 saal ke records mein kuch nahi mila. Ek baar sirf 'Abdul Kalam' bol kar dekhiye?"; } } } catch (e) { response = "Sir, internet connectivity mein issue hai."; } } else { response = "Aapne kuch kaha, par main samajh nahi paya."; } } speak(response); }
